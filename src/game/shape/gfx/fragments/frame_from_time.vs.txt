struct frameIndices {
    int index1;
    int index2;
    float t;
};

frameIndices get_frame_indices() {
    float time = timeData.x;
    frameIndices indices;

    indices.index1 = int(mod(time, 4.0)); 
    indices.index2 = indices.index1 + 1;

    if (indices.index2 >= 4) {
        indices.index2 = 0;
    }
    indices.t = mod(time, 1.0);

    return indices;
}
